15  package org.esigate.servlet.impl;
16  
17  import javax.servlet.http.HttpServletRequest;
18  
19  import org.esigate.impl.UriMapping;
20  import org.slf4j.Logger;
21  import org.slf4j.LoggerFactory;
22  
23  /**
24   * Utility class to work with Servlet request urls.
25   * 
26   * @author Nicolas Richeton
27   * 
28   */
29  public final class RequestUrl {
30      private static final Logger LOG = LoggerFactory.getLogger(RequestUrl.class);
31  
32      private RequestUrl() {
33  
34      }
35  
36      /**
37       * Get the relative url to the current servlet.
38       * <p>
39       * Uses the request URI and removes : the context path, the servlet path and the mapping path if used.
40       * 
41       * @param request
42       *            The current HTTP request
43       * @param mapping
44       *            matched mapping or null
45       * @param servlet 
46       * @return the url, relative to the servlet mapping.
47       */
48      public static String getRelativeUrl(HttpServletRequest request, UriMapping mapping, boolean servlet) {
49          // Raw request url
50          String relativeUrl = request.getRequestURI();
51          // Application (war) context path
52          String contextPath = request.getContextPath();
53          // Servlet mapping
54          String servletPath = request.getServletPath();
55          // Uri mapping
56          String mappingPath = (mapping == null ? null : mapping.getPath());
57  
58          if (LOG.isDebugEnabled()) {
59              LOG.debug("relativeUrl: {}, contextPath: {}, servletPath: {}, mappingPath: {}", new Object[] {relativeUrl,
60                      contextPath, servletPath, mappingPath});
61          }
62  
63          // Remove application context path
64          if (contextPath != null && relativeUrl.startsWith(contextPath)) {
65              relativeUrl = relativeUrl.substring(contextPath.length());
66          }
67  
68          // Remove servlet mapping path
69          if (servlet && servletPath != null && relativeUrl.startsWith(servletPath)) {
70              relativeUrl = relativeUrl.substring(servletPath.length());
71          }
72  
73          // Remove mapping path
74          if (mappingPath != null && relativeUrl.startsWith(mappingPath)) {
75              relativeUrl = relativeUrl.substring(mappingPath.length());
76          }
77  
78          return relativeUrl;
79      }
80  }
