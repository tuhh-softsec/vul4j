16  package org.esigate.servlet.impl;
17  
18  import java.io.IOException;
19  import java.util.Enumeration;
20  
21  import javax.servlet.FilterChain;
22  import javax.servlet.ServletContext;
23  import javax.servlet.ServletInputStream;
24  import javax.servlet.http.HttpServletRequest;
25  import javax.servlet.http.HttpServletResponse;
26  import javax.servlet.http.HttpSession;
27  
28  import org.apache.http.HttpHeaders;
29  import org.apache.http.ProtocolVersion;
30  import org.apache.http.entity.InputStreamEntity;
31  import org.apache.http.impl.cookie.BasicClientCookie;
32  import org.apache.http.message.BasicLineParser;
33  import org.apache.http.message.BasicRequestLine;
34  import org.esigate.http.IncomingRequest;
35  import org.esigate.servlet.HttpServletRequestContext;
36  import org.esigate.util.UriUtils;

38  /**
39   * Converts the {@link HttpServletRequest} to an {@link IncomingRequest}.
40   * 
41   * @author Francois-Xavier Bonnet
42   * 
43   */
44  public class RequestFactory {
45      private final ServletContext servletContext;
46  
47      public RequestFactory(ServletContext servletContext) {
48          this.servletContext = servletContext;
49      }
50  
51      public IncomingRequest create(HttpServletRequest request, HttpServletResponse response, FilterChain filterChain)
52              throws IOException {
53          HttpServletRequestContext context = new HttpServletRequestContext(request, response, servletContext,
54                  filterChain);
55          // create request line
56          String uri = UriUtils.createURI(request.getScheme(), request.getServerName(), request.getServerPort(),
57                  request.getRequestURI(), request.getQueryString(), null);
58          ProtocolVersion protocolVersion = BasicLineParser.parseProtocolVersion(request.getProtocol(), null);
59          IncomingRequest.Builder builder = IncomingRequest.builder(new BasicRequestLine(request.getMethod(), uri,
60                  protocolVersion));
61          builder.setContext(context);
62          // copy headers
63          @SuppressWarnings("rawtypes")
64          Enumeration names = request.getHeaderNames();
65          while (names.hasMoreElements()) {
66              String name = (String) names.nextElement();
67              @SuppressWarnings("rawtypes")
68              Enumeration values = request.getHeaders(name);
69              while (values.hasMoreElements()) {
70                  String value = (String) values.nextElement();
71                  builder.addHeader(name, value);
72              }
73          }
74          // create entity
75          ServletInputStream inputStream = request.getInputStream();
76          if (inputStream != null) {
77              // Copy entity-related headers
78              String contentLengthHeader = request.getHeader(HttpHeaders.CONTENT_LENGTH);
79              long contentLength = (contentLengthHeader != null) ? Long.parseLong(contentLengthHeader) : -1;
80              InputStreamEntity entity = new InputStreamEntity(inputStream, contentLength);
81              String contentTypeHeader = request.getContentType();
82              if (contentTypeHeader != null) {
83                  entity.setContentType(contentTypeHeader);
84              }
85              String contentEncodingHeader = request.getCharacterEncoding();
86              if (contentEncodingHeader != null) {
87                  entity.setContentEncoding(contentEncodingHeader);
88              }
89              builder.setEntity(entity);
90          }
91  
92          builder.setRemoteAddr(request.getRemoteAddr());
93          builder.setRemoteUser(request.getRemoteUser());
94          HttpSession session = request.getSession(false);
95          if (session != null) {
96              builder.setSessionId(session.getId());
97          }
98          builder.setUserPrincipal(request.getUserPrincipal());
99          // Copy cookies
100         javax.servlet.http.Cookie[] src = request.getCookies();
101         if (src != null) {
102             for (int i = 0; i < src.length; i++) {
103                 javax.servlet.http.Cookie c = src[i];
104                 BasicClientCookie dest = new BasicClientCookie(c.getName(), c.getValue());
105                 dest.setSecure(c.getSecure());
106                 dest.setDomain(c.getDomain());
107                 dest.setPath(c.getPath());
108                 dest.setComment(c.getComment());
109                 dest.setVersion(c.getVersion());
110                 builder.addCookie(dest);
111             }
112         }
113         builder.setSession(new HttpServletSession(request));
114         return builder.build();
115     }
